312 /************************************************************************/
313 /*>void blHPGLFont(int font, REAL size)
314  ------------------------------------
315 *//**
316 
317  \param[in] font Font number
318  \param[in] size Point size of font
319 
320  Set font for HPGL plot
321 
322 - 25.06.92 Taken from MoG
323 - 29.06.92 Corrected CA to CS and added SS. Changed to use global width
324  and height variables. Altered width to 1/2.4 * height
325 - 27.07.93 Changed precision of floating i/o to double
326 - 07.07.14 Use bl prefix for functions By: CTP
327 */
328 void blHPGLFont(int font,
329  REAL size)
330 {
331  char buffer[80];
332  REAL width;
333 
334  /* Character dimensions in centimeters. We convert pts to cm, then
335  divide by 1.7 to get a better size. Width is set to half height.
336  */
337  sFontHcm = size * 2.54 / (1.7 * 72.0);
338  sFontWcm = sFontHcm/2.4;
339 
340  sprintf(buffer,"PU;CS%d;SS;SI%5.3f, %5.3f;\n",
341  font,sFontWcm,sFontHcm);
342  fputs(buffer,sHPGLFile);
343 
344  /* Character height in scaled units */
345  sFontHeight = (int)((size * 10000.0) / (72.0 * PSypicsize));
346 
347  /* Character width from height in cm. This conversion is empirical! */
348  width = sFontHcm * 0.6154; /* cm */
349  width /= 2.54; /* in */
350  width *= 10000.0/PSxpicsize; /* Graph units */
351  sFontWidth = (int)width;
352 }

547 /************************************************************************/
548 /*>void blHPGLVText(REAL x, REAL y, REAL xoff, char *text, int TitleFont,
549  REAL TitleSize, char *label, int LabelFont, REAL LabelSize)
550  -----------------------------------------------------------------------
551 *//**
552 
553  \param[in] x X coordinate (in data units)
554  \param[in] y Y coordinate (in data units)
555  \param[in] xoff X-offset in pts
556  \param[in] *text Text to be written
557  \param[in] *TitleFont Font in which to write it
558  \param[in] TitleSize Size of font
559  \param[in] *label Label to be used to calc x offset
560  \param[in] *LabelFont Font of this label
561  \param[in] LabelSize Size of this label
562 
563  Write vertical text centred on x,y offset back along x by the size of
564  label and by xoff in pts
565  The `label' specification is used to calculate an amount by
566  which to move the text back. Typically this would be the longest data
567  label on the graph's Y-axis.
568  The FIXVERT code is used to correct character dimensions. Not sure if
569  it is a bug in the PLT: interpreter which requires this, or whether
570  it is standard HPGL. Define FIXVERT if it improves your output.
571 
572 - 25.06.92 Framework
573 - 29.06.92 Original
574 - 06.07.92 Modified x-pos for wider font width
575 - 27.07.93 Changed precision of floating i/o to double
576 - 07.07.14 Use bl prefix for functions By: CTP
577 */
578 void blHPGLVText(REAL x,
579  REAL y,
580  REAL xoff,
581  char *text,
582  int TitleFont,
583  REAL TitleSize,
584  char *label,
585  int LabelFont,
586  REAL LabelSize)
587 {
588  REAL LabelWidth;
589  char buffer[240];
590 
591 #ifdef FIXVERT
592  REAL width,
593  fixwidth,
594  height,
595  fixheight;
596 #endif
597 
598  /* Find size of label */
599  LabelWidth = strlen(blSimplifyText(label)) *
600  (LabelSize * 10000.0) / (2.0 * 72.0 * PSxpicsize);
601  /* Convert offset from points to plotter units */
602  xoff *= 10000.0 / (72.0 * PSxpicsize);
603 
604  /* Convert x & y to plotter coordinates */
605  x *= 10000.0;
606  y *= 10000.0;
607 
608  /* Modify the x-coordinate to account for the offsets */
609  x -= LabelWidth;
610  x += xoff;
611 
612  /* Now find the y-start to centre the string vertically */
613  y -= strlen(blSimplifyText(text)) * sFontWidth / 2;
614  y += sFontWidth/6;
615 
616 #ifdef FIXVERT
617  /* Character dimensions in centimeters. We convert pts to cm, then
618  divide by 1.7 to get a better size. Width is set to half height
619  */
620  height = TitleSize * 2.54 / (1.7 * 72.0);
621  width = height/2.0;
622 
623  fixwidth = width * PSxpicsize/PSypicsize;
624  fixheight = height * PSypicsize/PSxpicsize;
625 
626  sprintf(buffer,"PU;SI%5.3f, %5.3f;\n",fixwidth,fixheight);
627  fputs(buffer,sHPGLFile);
628 #endif
629 
630  /* Now output the text */
631  sprintf(buffer,"PU;PA%d,%d;DR0,1;",(int)x,(int)y);
632  fputs(buffer,sHPGLFile);
633  blHPGLShowText(text,1,(int)x,(int)y);
634  fputs("DR;\n",sHPGLFile);
635 
636 #ifdef FIXVERT
637  sprintf(buffer,"PU;SI%5.3f, %5.3f;\n",width,height);
638  fputs(buffer,sHPGLFile);
639 #endif
640 }

657 /************************************************************************/
658 /*>void blHPGLShowText(char *text, BOOL orientation, int XBase, int YBase)
659  -----------------------------------------------------------------------
660 *//**
661 
662  \param[in] *text Text to be displayed with control codes
663  \param[in] orientation TRUE=vertical, FALSE=horizontal
664  \param[in] XBase Position at which to start (device coords)
665  \param[in] YBase Position at which to start (device coords)
666 
667  Displays the text, raising or lowering as appropriate and selecting
668  alternate font where required. Used by the various ...Text() routines.
669 
670 - 29.06.92 Original based on PostScript version.
671 - 27.07.93 Changed precision of floating i/o to double
672 - 11.03.94 Changed orientation to BOOL
673 - 07.07.14 Use bl prefix for functions By: CTP
674 - 13.03.19 Doubled the size of OutBuff By: ACRM
675 */
676 void blHPGLShowText(char *text,
677  BOOL orientation,
678  int XBase,
679  int YBase)
680 {
681  char buffer[MAXBUFF],
682  OutBuff[MAXBUFF*2];
683  int i, j,
684  chcount,
685  first = TRUE;
686  REAL fixwidth,
687  fixheight;
688 
689  /* Walk along the string */
690  for(i=0, j=0, chcount=0; i<strlen(text) && j<MAXBUFF-1; i++)
691  {
692  switch(text[i])
693  {
694  case '\\': /* Should interpret next character as Greek */
695  /* Finish the current string */
696  if(j)
697  {
698  buffer[j] = '\0';
699  sprintf(OutBuff,"LB%s\\;",buffer);
700  fputs(OutBuff,sHPGLFile);
701  j = 0;
702  }
703  /* Output the next character as Greek */
704  sprintf(OutBuff,"SA;LB%c\\;SS;",text[++i]);
705  fputs(OutBuff,sHPGLFile);
706  chcount++;
707  break;
708  case '^': /* Should raise next character */
709  /* Finish the current string */
710  if(j)
711  {
712  buffer[j] = '\0';
713  sprintf(OutBuff,"LB%s\\;",buffer);
714  fputs(OutBuff,sHPGLFile);
715  j = 0;
716  }
717 
718  if(first)
719  {
720  if(orientation) chcount++;
721  first = FALSE;
722  }
723 
724  /* Move to the shifted position */
725  if(orientation) /* VERTICAL */
726  sprintf(OutBuff,"PU;PA%d,%d;", XBase-sFontHeight/2,
727  (int)(YBase+(PSxpicsize*chcount*sFontWidth/PSypicsize)));
728  else /* HORIZONTAL */
729  sprintf(OutBuff,"PU;PA%d,%d;", XBase + chcount*sFontWidth,
730  YBase+sFontHeight/2);
731  fputs(OutBuff,sHPGLFile);
732 
733  /* If necessary build string */
734  if(text[++i] == '{')
735  {
736  while(text[++i] != '}' && text[i] != '\0' && j<MAXBUFF-1)
737  {
738  buffer[j++] = text[i];
739  chcount++;
