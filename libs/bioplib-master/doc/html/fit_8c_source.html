129 BOOL blMatfit(COOR *x1, /* First coord array */
130  COOR *x2, /* Second coord array */
131  REAL rm[3][3], /* Rotation matrix */
132  int n, /* Number of points */
133  REAL *wt1, /* Weight array */
134  BOOL column) /* Column-wise output */
135 {
136  int i,j;
137  REAL umat[3][3];
138 
139 
140  if(n<2)
141  {
142  return(FALSE);
143  }
144 
145  if(wt1)
146  {
147  for(i=0;i<3;i++)
148  {
149  for(j=0;j<3;j++) umat[i][j] = 0.0;
150 
151  for(j=0;j<n;j++)
152  {
153  switch(i)
154  {
155  case 0:
156  umat[i][0] += wt1[j] * x1[j].x * x2[j].x;
157  umat[i][1] += wt1[j] * x1[j].x * x2[j].y;
158  umat[i][2] += wt1[j] * x1[j].x * x2[j].z;
159  break;
160  case 1:
161  umat[i][0] += wt1[j] * x1[j].y * x2[j].x;
162  umat[i][1] += wt1[j] * x1[j].y * x2[j].y;
163  umat[i][2] += wt1[j] * x1[j].y * x2[j].z;
164  break;
165  case 2:
166  umat[i][0] += wt1[j] * x1[j].z * x2[j].x;
167  umat[i][1] += wt1[j] * x1[j].z * x2[j].y;
168  umat[i][2] += wt1[j] * x1[j].z * x2[j].z;
169  break;
170  }
171  }
172  }
173  }
174  else
175  {
176  for(i=0;i<3;i++)
177  {
178  for(j=0;j<3;j++) umat[i][j] = 0.0;
179 
180  for(j=0;j<n;j++)
181  {
182  switch(i)
183  {
184  case 0:
185  umat[i][0] += x1[j].x * x2[j].x;
186  umat[i][1] += x1[j].x * x2[j].y;
187  umat[i][2] += x1[j].x * x2[j].z;
188  break;
189  case 1:
190  umat[i][0] += x1[j].y * x2[j].x;
191  umat[i][1] += x1[j].y * x2[j].y;
192  umat[i][2] += x1[j].y * x2[j].z;
193  break;
194  case 2:
195  umat[i][0] += x1[j].z * x2[j].x;
196  umat[i][1] += x1[j].z * x2[j].y;
197  umat[i][2] += x1[j].z * x2[j].z;
198  break;
199  }
200  }
201  }
202  }

226 static void qikfit(REAL umat[3][3],
227  REAL rm[3][3],
228  BOOL column)
229 {
230 
231  REAL rot[3][3],
232  turmat[3][3],
233  c[3][3],
234  coup[3],
235  dir[3],
236  step[3],
237  v[3],
238  rtsum,rtsump,
239  stp,stcoup,
240  ud,tr,ta,cs,sn,ac,
241  delta,
242  gfac,
243  cle,
244  clep = 0.0;
245  int i,j,k,l,m,
246  jmax,
247  ncyc,
248  nsteep,
249  nrem;
250 
251  for(i=0; i<3; i++)
252  step[i] = 0.0;
253 
254  /* Rotate repeatedly to reduce couple about initial direction to zero.
255  Clear the rotation matrix
256  */
257  for(l=0;l<3;l++)
258  {
259  for(m=0;m<3;m++)
260  rot[l][m] = 0.0;
261  rot[l][l] = 1.0;
262  }
263 
264  /* Copy vmat[][] (sp) into umat[][] (dp) */
265  jmax = 30;
266  rtsum = umat[0][0] + umat[1][1] + umat[2][2];
267  delta = 0.0;
268 
269  for(ncyc=0;ncyc<jmax;ncyc++)
270  {
271  /* Modified CG. For first and every NSTEEP cycles, set previous
272  step as zero and do an SD step
273  */
274  nsteep = 3;
275  nrem = ncyc-nsteep*(int)(ncyc/nsteep);
276 
277  if(!nrem)
278  {
279  for(i=0;i<3;i++) step[i]=0.0;
280  clep = 1.0;
281  }
282 
283  /* Couple */
284  coup[0] = umat[1][2]-umat[2][1];
285  coup[1] = umat[2][0]-umat[0][2];
286  coup[2] = umat[0][1]-umat[1][0];
287  cle = sqrt(coup[0]*coup[0] + coup[1]*coup[1] + coup[2]*coup[2]);
288 
289  /* Gradient vector is now -coup */
290  gfac = (cle/clep)*(cle/clep);
291 
292  /* Value of rtsum from previous step */
293  rtsump = rtsum;
294  clep = cle;
295  if(cle < SMALL) break;
296 
297  /* Step vector conjugate to previous */
298  stp = 0.0;
299  for(i=0;i<3;i++)
300  {
301  step[i]=coup[i]+step[i]*gfac;
302  stp += (step[i] * step[i]);
303  }
304  stp = 1.0/sqrt(stp);
305 
306  /* Normalised step */
307  for(i=0;i<3;i++) dir[i] = stp*step[i];
308 
309  /* Couple resolved along step direction */
310  stcoup = coup[0]*dir[0] + coup[1]*dir[1] + coup[2]*dir[2];
311 
312  /* Component of UMAT along direction */
313  ud = 0.0;
314  for(l=0;l<3;l++)
315  for(m=0;m<3;m++)
316  ud += umat[l][m]*dir[l]*dir[m];
317 
318 
319  tr = umat[0][0]+umat[1][1]+umat[2][2]-ud;
320  ta = sqrt(tr*tr + stcoup*stcoup);
321  cs=tr/ta;
322  sn=stcoup/ta;
323 
324  /* If cs<0 then posiiton is unstable, so don't stop */
325  if((cs>0.0) && (ABS(sn)<SMALSN)) break;
326 
327  /* Turn matrix for correcting rotation:
328 
329  Symmetric part
330  */
331  ac = 1.0-cs;
332  for(l=0;l<3;l++)
333  {
334  v[l] = ac*dir[l];
335  for(m=0;m<3;m++)
336  turmat[l][m] = v[l]*dir[m];
337  turmat[l][l] += cs;
338  v[l]=dir[l]*sn;
339  }
340 
341  /* Asymmetric part */
342  turmat[0][1] -= v[2];
343  turmat[1][2] -= v[0];
344  turmat[2][0] -= v[1];
345  turmat[1][0] += v[2];
346  turmat[2][1] += v[0];
347  turmat[0][2] += v[1];
348 
349  /* Update total rotation matrix */
350  for(l=0;l<3;l++)
351  {
352  for(m=0;m<3;m++)
353  {
354  c[l][m] = 0.0;
355  for(k=0;k<3;k++)
356  c[l][m] += turmat[l][k]*rot[k][m];
357  }
358  }
359 
360  for(l=0;l<3;l++)
361  for(m=0;m<3;m++)
362  rot[l][m] = c[l][m];
363 
364  /* Update umat tensor */
365  for(l=0;l<3;l++)
366  for(m=0;m<3;m++)
367  {
368  c[l][m] = 0.0;
369  for(k=0;k<3;k++)
370  c[l][m] += turmat[l][k]*umat[k][m];
371  }
372 
373  for(l=0;l<3;l++)
374  for(m=0;m<3;m++)
375  umat[l][m] = c[l][m];
376 
377  rtsum = umat[0][0] + umat[1][1] + umat[2][2];
378  delta = rtsum - rtsump;
379 
380  /* If no improvement in this cycle then stop */
381  if(ABS(delta)<SMALL) break;
382 
383  /* Next cycle */
384  }
385 
386  /* Copy rotation matrix for output */
387  if(column)
388  {
389  for(i=0;i<3;i++)
390  for(j=0;j<3;j++)
391  rm[j][i] = rot[i][j];
392  }
393  else
394  {
395  for(i=0;i<3;i++)
396  for(j=0;j<3;j++)
397  rm[i][j] = rot[i][j];
398  }
399 }
