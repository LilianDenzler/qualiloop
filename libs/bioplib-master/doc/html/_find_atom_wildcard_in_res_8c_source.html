100 PDB *blFindAtomWildcardInRes(PDB *pdb, char *pattern)
101 {
102  PDB *p, *pNext;
103  int i;
104  BOOL ok;
105 
106  pNext = blFindNextResidue(pdb);
107 
108  for(p=pdb; p!=pNext; NEXT(p))
109  {
110  if(!strncmp(p->atnam,pattern,4))
111  return(p);
112  ok = TRUE;
113  for(i=0; i<4; i++)
114  {
115  if((pattern[i] != p->atnam[i]) && pattern[i] != '?')
116  {
117  ok = FALSE;
118  break;
119  }
120  }
121  if(ok)
122  return(p);
123  }
124  return(NULL);
125 }
