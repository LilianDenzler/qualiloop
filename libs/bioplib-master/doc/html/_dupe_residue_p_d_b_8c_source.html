100 PDB *blDupeResiduePDB(PDB *in)
101 {
102  PDB *pNext,
103  *out = NULL,
104  *p = NULL,
105  *q = NULL;
106 
107  /* Find the next residue */
108  pNext = blFindNextResidue(in);
109 
110  for(p=in; p!=pNext; NEXT(p))
111  {
112  if(out==NULL)
113  {
114  INIT(out, PDB);
115  q=out;
116  }
117  else
118  {
119  ALLOCNEXT(q, PDB);
120  }
121  if(q==NULL)
122  {
123  FREELIST(out, PDB);
124  return(NULL);
125  }
126 
127  blCopyPDB(q, p);
128  q->nConect=0;
129  }
130 
131  return(out);
132 }
