85 
86 
87  Usage:
88  ======
89 
90  The deprecated.h header file is included at the end of a header file
91  with deprecated functions. This is to ensure that data structures are
92  defined _before_ the prototypes that use them are declared in
93  deprecated.h. For example, the PDB data structure must be defined
94  before declaring the prototypes for the functions returning a PDB
95  linked list.
96 
97  Prototypes for the deprecated functions for each header file are
98  enclosed within compiler directives to ensure that functions are not
99  declared using undefined data structures.

139 /*>DEPRECATED(s, t)
140  ----------------
141 *//**
142 
143  The DEPRECATED macro gives a warning message if a function is
144  deprecated and indicates the replacement function.
145 
146  The default option is to give a warning message unless an environment
147  variable, BIOPLIB_DEPRECATED_QUIET, is set.
148 
149  Alternatively the compile options: -D BIOPLIB_DEPRECATED_CHECK or
150  -D BIOPLIB_DEPRECATED_QUIET will set the DEPRECATED macro to ignore the
151  BIOPLIB_DEPRECATED_QUIET environment variable.
152 
153  -D BIOPLIB_DEPRECATED_CHECK will display the warning message.
154  -D BIOPLIB_DEPRECATED_QUIET will always silence the warning message.
155 
156 - 29.04.14 Original By: CTP
157 */
158 # if(defined BIOPLIB_DEPRECATED_CHECK || defined BIOPLIB_DEPRECATED_QUIET)
159 # ifndef BIOPLIB_DEPRECATED_QUIET
160 # define DEPRECATED(s, t) \
161  fprintf(stderr, \
162  "This code uses %s which is now deprecated!\n" \
163  " Use %s instead\n", (s), (t))
164 # else
165 # define DEPRECATED(s, t)
166 # endif
167 # else
168 # define DEPRECATED(s, t) \
169  { \
170  if(!getenv("BIOPLIB_DEPRECATED_QUIET")) \
171  fprintf(stderr, \
172  "This code uses %s which is now deprecated!\n" \
173  " Use %s instead\n" \
174  " Set the BIOPLIB_DEPRECATED_QUIET environment \
175 variable to silence this message\n" , (s), (t)); \
176  }
177 # endif
