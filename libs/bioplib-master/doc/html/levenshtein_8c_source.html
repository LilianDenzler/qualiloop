78 /*>int blLevenshteinDistance(char *columnString, char *rowString)
79  --------------------------------------------------------------
80 *//**
81  \input columnString A NULL-terminated string pointer
82  \input rowString A NULL-terminated string pointer
83  \return Levenshtein distance (<0 for error)
84 
85  Calculates the Levenshtein edit distance using the iterative
86  method with two matrix rows. See
87  https://en.m.wikipedia.org/wiki/Levenshtein_distance
88 
89  The method is memory efficient as it uses only two rows from the
90  complete matrix - to minimize memory usage, ensure that rowString
91  is the shorter string.
92 
93  However, it is suboptimal. The amount of memory required may be
94  reduced to one row and one (index) word of overhead, for better
95  cache locality. Thus speed comes from using very little memory,
96  often keeping the buffer entirely in cache, and reducing the amount
97  of work by skipping any prefix and postfix that won't add to the
98  cost. You only really need to know the three values when you want
99  to update a cell in the matrix and you can keep two of them in a
100  buffer, while keeping the third value in a fixed location.
101 
102  See
103  https://web.archive.org/web/20180612143641/https://bitbucket.org/clearer/iosifovich/
104 
105  and the C++ implementation of that version at
106  https://github.com/rljacobson/Levenshtein/blob/master/damlev.cpp
107 
108 - 15.06.20 Original By: ACRM
109 */
110 int blLevenshteinDistance(char *columnString, char *rowString)
111 {
112  int columnSize = strlen(columnString);
113  int rowSize = strlen(rowString);
114  int *rowPrevious = NULL;
115  int *rowCurrent = NULL;
116  int i, j, result;
117 
118  /* Allocate memory for the two rows */
119  if((rowPrevious = (int *)malloc((rowSize+1) * sizeof(int)))==NULL)
120  return(-1);
121 
122  if((rowCurrent = (int *)malloc((rowSize+1) * sizeof(int)))==NULL)
123  {
124  free(rowPrevious);
125  return(-1);
126  }
127 
128  /* Initialize rowPrevious (the previous row of distances). This
129  row is A[0][i] (where A[][] would be the full matrix). The edit
130  distance for an empty columnString is just the number of
131  characters to delete from rowString
132  */
133  for(i=0; i<=rowSize; i++)
134  {
135  rowPrevious[i] = i;
136  }
137 
138  for(i=0; i<columnSize; i++)
139  {
140  /* Calculate rowCurrent (current row distances) from the
141  previous row (rowPrevious). The first element of rowCurrent
142  is A[i+1][0], the edit distance comes from deleting (i+1)
143  chars from columnString to match an empty rowString
144  */
145  rowCurrent[0] = i + 1;
146 
147  /* use the formula to fill in the rest of the row */
148  for(j=0; j<rowSize; j++)
149  {
150  /* Calculate costs for A[i+1][j+1] */
151  int deletionCost = rowPrevious[j+1] + 1;
152  int insertionCost = rowCurrent[j] + 1;
153  int substitutionCost = 0;
154 
155  if(columnString[i] == rowString[j])
156  {
157  substitutionCost = rowPrevious[j];
158  }
159  else
160  {
161  substitutionCost = rowPrevious[j] + 1;
162  }
163 
164  rowCurrent[j+1] = MIN3(deletionCost,
165  insertionCost,
166  substitutionCost);
167  }
168 
169  /* Copy rowCurrent (current row) to rowPrevious (previous row)
170  for the next iteration. Since the data in rowCurrent are
171  always invalidated, for efficiency, the swap is done just by
172  swapping the pointers.
173  */
174  SWAPINTPTR(rowPrevious, rowCurrent);
175  }
176 
177  /* After the final swap, the results of rowCurrent are now in
178  rowPrevious
179  */
180  result = rowPrevious[rowSize];
181  free(rowPrevious);
182  free(rowCurrent);
183 
184  return(result);
185 }
