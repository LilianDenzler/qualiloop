139  char *insert1, char *chain2, int resnum2,

141  -----------------------------------------------------------------------

146  \param[in] resnum1 Start residue number
147  \param[in] *insert1 Start residue insert code

150  \param[in] *insert2 End residue insert code
151  \return PDB linked list of the region of interest.
152 
153  Reduces a PDB linked list to those residues within a specified zone.
154  Note that the PDB linked list is duplicated before extraction so
155  pointers do not match those in the input PDB linked list. Excess
156  records in the new PDB linked list are freed.
157 
158 - 10.01.96 Original By: ACRM
159 - 22.03.06 Modified to allow non-exact zones. i.e. the extracted zone
160  will be the widest subset of the specified zone. So, if
161  you specifiy 30-35Z and the PDB file only has 30-35B
162  then that will be extracted.
163 - 29.10.10 Fixed extraction where end of zone matched last residue in

170 - 17.02.16 Added fix on check for exact match - prev wasn't being set
171  correctly if the match was to the first residue in the file
172 
173 *** TODO - This doesn't deal with CONECT information properly! ***
174 */
175 PDB *blExtractZonePDBAsCopy(PDB *inpdb,
176  char *chain1, int resnum1, char *insert1,
177  char *chain2, int resnum2, char *insert2)
178 {
179  PDB *pdb, *p,
180  *start = NULL,
181  *last = NULL,
182  *prev = NULL;
183 
184  /* Duplicate the PDB linked list */
185  if((pdb = blDupePDB(inpdb))==NULL)
186  return(NULL);
187 
188  /* Try to find the exact residue specification, setting start to the
189  start of that residue and prev to the last atom of the previous
190  residue
191  */
192  if((start=blFindResidue(pdb, chain1, resnum1, insert1))!=NULL)
193  {
194  /* Find the previous atom */
195  if(start==pdb)
196  {
197  prev=NULL;
198  }
199  else
200  {
201  for(prev=pdb; prev->next!=start; NEXT(prev)) ;
202  }
203 
204  }
205  else
206  {
207  /* We didn't find an exact match so find the residue that would
208  follow that residue specification.
209  Again prev will point to the last atom before the first atom in
210  the zone and start will point to the first atom in the zone
211  */
212  for(p=pdb; p!=NULL; NEXT(p))
213  {
214  if(CHAINMATCH(p->chain,chain1) &&
215  ((p->resnum > resnum1) ||
216  ((p->resnum == resnum1) &&
217  (p->insert[0] >= insert1[0]))))
218  {
219  start = p;
220  break;
221  }
222  prev = p;
223  }
224  }
225 
226  /* If we didn't find anything then exit */
227  if(start==NULL)
228  {
229  FREELIST(pdb, PDB);
230  return(NULL);
231  }
232 
233 
234  /* See if we have an exact match to the residue requested
235  last will be the last atom in that residue
236  */
237  if((last=blFindResidue(pdb, chain2, resnum2, insert2))!=NULL)
238  {
239  PDB *nextres = blFindNextResidue(last);
240 
241  /* Step to the end of this residue */
242  for(; last->next!=nextres; NEXT(last)) ;
243  }
244  else
245  {
246  /* There was no exact match so find the one after that residue
247  specification.
248  Again, last will point to the last atom in the zone
249 
250  29.10.10 Also breaks out if chain1 and chain2 are the same but
251  we've now come to a different chain. This fixes a bug where
252  the code wouldn't break out if resnum2 was the last residue in
253  a chain By: ACRM
254  */
255  for(p=start; p!=NULL; NEXT(p))
256  {
257  if(CHAINMATCH(p->chain,chain2) && /* If same chain and... */
258  ((p->resnum > resnum2) || /* Residue number exceeded or*/
259  ((p->resnum == resnum2) && /* Resnum same and... */
260  (p->insert[0] > insert2[0])))) /* insert exceeded */
261  {
262  break;
263  }
264  /* Both zone ends are in the same chain so, if we got here we have
265  found the right chain. If the current chain is now different
266  from chain2, then we've gone off the end of the chain
267  */
268  if( CHAINMATCH(chain1,chain2) &&
269  !CHAINMATCH(p->chain,chain2))
270  {
271  break;
272  }
273  last = p;
274  }
275  }
276 
277  if(last==NULL)
278  {
279  FREELIST(pdb, PDB);
280  return(NULL);
281  }
282 
283  /* Free linked list after 'last' */
284  if(last->next != NULL)
285  {
286  FREELIST(last->next, PDB);
287  last->next = NULL;
288  }
289 
290  /* Unlink 'start' from rest of linked list and free memory before
291  'start'
292  */
293  if(prev != NULL)
294  {
295  prev->next = NULL;
296  FREELIST(pdb, PDB);
297  }
298 
299  return(start);
300 }

483 PDB *blForceExtractNotZonePDBAsCopy(PDB *inpdb,
484  char *chain1, int resnum1,
485  char *insert1,
486  char *chain2, int resnum2,

488 {

490  *start = NULL,

492  *p, *q;
493 
494  /* Find the start and stop of the zone in the old linked list */
495  start = blFindResidue(inpdb, chain1, resnum1, insert1);
496  stop = blFindResidue(inpdb, chain2, resnum2, insert2);
497 
498  if((start != NULL) && (stop != NULL))
499  {
500  return(blExtractNotZonePDBAsCopy(inpdb,
501  chain1, resnum1, insert1,
502  chain2, resnum2, insert2));
503  }
504  else if((start == NULL) && (stop == NULL))
505  {
506  /* Neither residue found so simply return the original PDB
507  linked list
508  */
509  return(blDupePDB(inpdb));
510  }
511  else if(start == NULL)
512  {
513  stop = blFindNextResidue(stop);
514 
515  for(p=inpdb; p!=NULL; NEXT(p))
516  {
517  /* If the chain is different, just copy it */
518  if(!CHAINMATCH(p->chain, chain1))
519  {
520  if(pdb == NULL)
521  {
522  INIT(pdb, PDB);
523  q = pdb;
524  }
525  else
526  {
527  ALLOCNEXT(q, PDB);
528  }
529 
530  if(q==NULL)
531  {
532  FREELIST(pdb, PDB);
533  return(NULL);
534  }
535 
536  blCopyPDB(q, p);
537  }
538  else /* Chain is the same - copy from 'stop' onwards */
539  {
540  PDB *r, *prev=NULL;
541 
542  for(r=stop; r!=NULL; NEXT(r))
543  {
544  if(pdb == NULL)
545  {
546  INIT(pdb, PDB);
547  q = pdb;
548  }
549  else

551  ALLOCNEXT(q, PDB);
552  }
553

562  }
563 
564  p=prev; /* Force finish the outer loop */
565  } /* End of chain is the same */
566  } /* End of inpdb loop */
567  } /* End of start==NULL, stop!=NULL */
568  else /* start!=NULL; stop==NULL */
569  {
570  for(p=inpdb; p!=NULL; NEXT(p))
571  {
572  /* If the chain is different, just copy it */
573  if(!CHAINMATCH(p->chain, chain2))
574  {
575  if(pdb == NULL)
576  {
577  INIT(pdb, PDB);
578  q = pdb;
579  }

589  }
590

592  }
593  else
594  {
595  PDB *r;
596  for(r=p; r!=start; NEXT(r))
597  {
598  if(pdb == NULL)

618

622  {
623  r=r->next;
624  }
625 
626  p=r;
627  } /* It's the chain of interest */
628  } /* Outer step through linked list */
629  } /* End of start!=NULL; stop==NULL */
