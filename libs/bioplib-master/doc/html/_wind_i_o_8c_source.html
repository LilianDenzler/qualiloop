388 int blYorN(char deflt)
389 {
390  char buffer[80],
391  response;
392  int i;
393 
394  blGetKybdString(buffer, 20);
395  response = buffer[0];
396 
397  if((response != 'Y') && (response != 'y') &&
398  (response != 'N') && (response != 'n') &&
399  (response != 'A') && (response != 'a') &&
400  (response != 'Q') && (response != 'q'))
401  {
402  response = deflt;
403  }
404 
405  switch(response)
406  {
407  case 'Y':
408  case 'y':
409  i = 1;
410  break;
411  case 'N':
412  case 'n':
413  i = 0;
414  break;
415  case 'A':
416  case 'a':
417  i = 2;
418  break;
419  case 'Q':
420  case 'q':
421  i = 3;
422  break;
423  default: /* Should never occur */
