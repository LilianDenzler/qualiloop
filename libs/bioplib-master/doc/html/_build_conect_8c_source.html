193  \param[in,out] *q Second PDB item
194  \return Success?

201 BOOL blAddOneDirectionConect(PDB *p, PDB *q)
202 {
203  int i;

205 
206  /* See if we have the conect already */
207  gotConect=FALSE;
208  for(i=0; i<p->nConect; i++)
209  {
210  if(p->conect[i] == q)
211  {
212  gotConect = TRUE;

218  if(!gotConect)

229  }

234 /************************************************************************/
235 /*>BOOL blBuildConectData(PDB *pdb, REAL tol)
236  ------------------------------------------
237 *//**
238  \param[in,out] *pdb PDB linked list
239  \param[in] tol Tolerence for distance between atoms
240  \return Were all CONECTs added OK
241 
242  Deletes all current connectivity data and rebuilds it using covalent
243  radii data. A return of FALSE indicates that there were too many
244  connections for an atom. If this happens, MAXCONECT needs to be
245  increased in pdb.h
246 
247 - 19.02.15 Original By: ACRM
248 - 26.02.15 Added tol paramater
249 - 12.05.15 Conects are built involving backbone C and N if either atom
250  is a HETATM
251 */
252 BOOL blBuildConectData(PDB *pdb, REAL tol)
253 {
254  PDB *p,
255  *q,
256  *res,
257  *nextRes;
258  BOOL retval=TRUE;
259 
260  /* Clear all current connect data */
261  for(p=pdb; p!=NULL; NEXT(p))
262  {
263  p->nConect = 0;
264  }
265 
266  for(res=pdb; res!=NULL; res=nextRes)
267  {
268  nextRes = blFindNextResidue(res);
269 
270  /* Check for any HETATM connections within this residue */
271  for(p=res; p!=nextRes; NEXT(p))
272  {
273  for(q=p->next; q!=nextRes; NEXT(q))
274  {
275  if(!strncmp(p->record_type, "HETATM", 6) ||
276  !strncmp(q->record_type, "HETATM", 6))
277  {
278  if(blIsBonded(p, q, tol))
279  {
280  if(!blAddConect(p,q))
281  retval=FALSE;
282  }
283  }
284  }
285  }
286 
287  /* Check for connections between residues which don't involve backbone
288  C or N, or do involve HETATMS
289  */
290  for(p=res; p!=nextRes; NEXT(p))
291  {
292  for(q=nextRes; q!=NULL; NEXT(q))
293  {
294  if(strncmp(p->atnam, "C ", 4) ||
295  strncmp(q->atnam, "N ", 4) ||
296  !strncmp(p->record_type, "HETATM", 6) ||
297  !strncmp(q->record_type, "HETATM", 6))
298  {
299  if(blIsBonded(p, q, tol))
300  {
301  if(!blAddConect(p,q))
302  retval=FALSE;
303  }
304  }
305  }
306  }
307  }

320  \return Bonded?
321 
322  Test whether two atoms are bonded
323 
324 - 19.02.15 Original By: ACRM
325 - 26.02.15 Added tol parameter. Changed to used squared distances
326 */
327 BOOL blIsBonded(PDB *p, PDB *q, REAL tol)
328 {
329  REAL r1, r2, bondDist;
330  r1 = findCovalentRadius(p->element);
331  r2 = findCovalentRadius(q->element);
332  bondDist = (r1+r2+tol);
333 
334  if(DISTSQ(p,q) <= bondDist*bondDist)
335  return(TRUE);
336  return(FALSE);
337 }

488 /*>BOOL blDeleteAConectByNum(PDB *pdb, int cNum)
489  ------------------------------------------
490 *//**
491  \param[in] *pdb PDB pointer
492  \param[in] cNum Index into the ->conect[] array for the CONECT
493  to be deleted
494  \return Success
495 
496  Deletes the link for the specified CONECT. Other CONECTs are shuffled
497  down.
498 
499 - 16.03.15 Original By: ACRM
500 - 29.08.18 Added check on MAXCONECT
501 */
502 BOOL blDeleteAConectByNum(PDB *pdb, int cNum)
503 {
504  int i;
505 
506  /* Check that the CONECT exists */
507  if((cNum >= pdb->nConect) || (pdb->nConect == 0))
508  return(FALSE);
509 
510  /* Shuffle the CONECTs down */
511  for(i=cNum; i<pdb->nConect; i++)
512  {
513  PDB *next = NULL;
514  if(((i+1) < pdb->nConect) &&
515  ((i+1) < MAXCONECT))
516  next = pdb->conect[i+1];
517 
518  pdb->conect[i] = next;
519  }
520 
521  /* Decrement the number of CONECTs and return */
522  pdb->nConect--;
523  return(TRUE);
524 }

561 /************************************************************************/
562 /*>BOOL blCopyConects(PDB *out, PDB *in)
563  -------------------------------------
564 *//**
565  \param[out] *out PDB linked list containing new CONECT records
566  \param[in] *in PDB linked list containing original CONECT records
567  \return Success
568 
569  Updates the CONECT information in the out PDB linked list based on
570  those in the in linked list. This is used when we have copied a PDB
571  linked list (e.g. in the bl...AsCopy() routines) to make sure the
572  CONECT data points to records in the new linked list instead of the
573  old one.
574 
575 - 17.04.15 Original By: ACRM
576 */
577 BOOL blCopyConects(PDB *out, PDB *in)
578 {
579  PDB **idxOut = NULL,
580  *p;
581  int indexSize = 0;
582 
583  /* Create an indices by atom number */
584  if((idxOut = blIndexAtomNumbersPDB(out, &indexSize))==NULL)
585  return(FALSE);
586 
587  /* Step through the output linked list */
588  for(p=out; p!=NULL; NEXT(p))
589  {
590  int i;
591 
592  /* Step through the CONECT data for this atom - which will be
593  from the OLD linked list
594  */
595  for(i=0; i<p->nConect; i++)
596  {
597  if(p->conect[i] != NULL)
598  {
599  int atomNum;
600  PDB *newPtr = NULL;
601 
602  /* Find the atom number from old linked list */
603  atomNum = (p->conect[i])->atnum;
604 
605  /* Look up the pointer for this in the new linked list index*/
606  if(atomNum < indexSize)
607  newPtr = idxOut[atomNum];
608 
609  /* Reset pointer */
610  p->conect[i] = newPtr;
611  }
612  }
613  }
614 
615  free(idxOut);
616  return(TRUE);
617 }
